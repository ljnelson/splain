050    /**
051     * A factory for localized messages appropriate for object graphs.
052     *
053     * @param <T> the type of {@link Object} used by the {@link
054     * #getMessage(List)} method; the type of {@link Object} used by the
055     * {@link Pattern}s that help select messages
056     *
057     * @author <a href="http://about.me/lairdnelson"
058     * target="_parent">Laird Nelson</a>
059     */
060    public class MessageFactory<T> implements Serializable {
061      
062      /**
063       * The version of this class for {@linkplain Serializable
064       * serialization purposes}.
065       */
066      private static final long serialVersionUID = 1L;

069       * A {@link Map} of {@link Set}s of {@link Pattern}s, indexed by

072       * <p>This field may be {@code null}.</p>
073       */
074      private Map<ResourceBundleKey, Set<Pattern<T>>> patterns;

077       * Creates a new {@link MessageFactory}.

083      /**
084       * Adds a {@link Pattern} to the {@link Set} of {@link Pattern}s
085       * indexed under the supplied {@link ResourceBundleKey}.  If no such
086       * {@link Set} exists, one is created whose sole contents are the
087       * supplied {@link Pattern}.
088       *
089       * @param key the {@link ResourceBundleKey} under which the supplied
090       * {@link Pattern} is to be indexed; must not be {@code null}
091       *
092       * @param pattern the {@link Pattern} to add; must not be {@code
093       * null}
094       *
095       * @return {@code true} if and only if the supplied {@link Pattern}
096       * was indexed under the supplied {@link ResourceBundleKey}; {@code
097       * false} if, for example, it was already present
098       *
099       * @exception IllegalArgumentException if {@code key} or {@code
100       * pattern} is {@code null}
101       *
102       * @see Pattern
103       *
104       * @see ResourceBundleKey
105       *
106       * @see #addPatterns(ResourceBundleKey, Iterable)
107       */
108      public final boolean addPattern(final ResourceBundleKey key, final Pattern<T> pattern) {
109        if (key == null) {
110          throw new IllegalArgumentException("key", new NullPointerException("key"));
111        }
112        if (pattern == null) {
113          throw new IllegalArgumentException("pattern", new NullPointerException("patterns"));
114        }
115        return this.addPatterns(key, Collections.singleton(pattern));
116      }
117    
118      /**
119       * Adds the supplied {@link Iterable} of {@link Pattern}s to the
120       * {@link Set} of {@link Pattern}s indexed under the supplied {@code
121       * key}.
122       *
123       * @param key the {@link ResourceBundleKey} under which the {@link
124       * Pattern} is to be stored; must not be {@code null}
125       *
126       * @param patterns an {@link Iterable} of {@link Pattern}s to add;
127       * no references are kept to this object; must not be {@code null}
128       *
129       * @return {@code true} if the set of {@link Pattern}s already
130       * indexed under the supplied {@code key}&mdash;if any&mdash;was
131       * changed as a result of this call; {@code false} otherwise
132       *
133       * @exception IllegalArgumentException if either parameter is {@code
134       * null}
135       *
136       * @see Pattern
137       *
138       * @see ResourceBundleKey
139       */
140      public boolean addPatterns(final ResourceBundleKey key, final Iterable<Pattern<T>> patterns) {
141        if (key == null) {
142          throw new IllegalArgumentException("key", new NullPointerException("key"));
143        }
144        if (patterns == null) {
145          throw new IllegalArgumentException("patterns", new NullPointerException("patterns"));
146        }
147        if (this.patterns == null) {
148          this.patterns = new LinkedHashMap<ResourceBundleKey, Set<Pattern<T>>>();
149        }
150        Set<Pattern<T>> patternSet = this.patterns.get(key);
151        if (patternSet == null) {
152          patternSet = new LinkedHashSet<Pattern<T>>();
153          this.patterns.put(key, patternSet);
154        }
155        boolean returnValue = false;
156        for (final Pattern<T> pattern : patterns) {
157          if (pattern != null) {
158            returnValue = patternSet.add(pattern) || returnValue;
159          }
160        }
161        return returnValue;
162      }
163    
164      /**
165       * Returns a non-{@code null}, {@linkplain
166       * Collections#unmodifiableSet(Set) unmodifiable <code>Set</code> view}
167       * of {@link Pattern}s indexed under the supplied {@code key}.
168       *
169       * <p>This method never returns {@code null}.</p>
170       *
171       * @param key the {@link ResourceBundleKey} whose {@link Pattern}s
172       * are to be retrieved; must not be {@code null}
173       *
174       * @return a non-{@code null}, {@linkplain
175       * Collections#unmodifiableSet(Set) unmodifiable <code>Set</code> view}
176       * of {@link Pattern}s indexed under the supplied {@code key}
177       *
178       * @exception IllegalArgumentException if {@code key} is {@code
179       * null}
180       */
181      public Set<Pattern<T>> getPatterns(final ResourceBundleKey key) {
182        if (key == null) {
183          throw new IllegalArgumentException("key", new NullPointerException("key"));
184        }
185        final Set<Pattern<T>> returnValue;
186        if (this.patterns == null || this.patterns.isEmpty() || !this.patterns.containsKey(key)) {
187          returnValue = Collections.emptySet();
188        } else {
189          final Set<Pattern<T>> set = this.patterns.get(key);
190          if (set == null || set.isEmpty()) {
191            returnValue = Collections.emptySet();
192          } else {
193            returnValue = Collections.unmodifiableSet(set);
194          }
195        }
196        return returnValue;
197      }

199      /**
200       * Formats or transforms the supplied {@code rawMessage} {@link
201       * Object} in some way, perhaps by using the information stored as
202       * part of the supplied {@link Matcher}, and returns the formatted
203       * or transformed {@link Object}.
204       *
205       * <p>This implementation checks to see if the supplied {@code
206       * rawMessage} is an instance of {@link CharSequence}.  If so, it is
207       * treated as an <a href="http://mvel.codehaus.org/">MVEL</a> <a
208       * href="http://mvel.codehaus.org/Templating+Guide">template</a>.
209       * The template is interpolated using all the {@linkplain
210       * Matcher#group(int) capture groups} and {@linkplain
211       * Matcher#getVariables() variables} that the supplied {@link
212       * Matcher} is capable of providing.</p>
213       *
214       * <p>Specifically, <a
215       * href="http://mvel.codehaus.org/MVEL+2.0+Orb+Tags">orb tags</a>
216       * may have bodies that reference the {@linkplain
217       * Matcher#getVariables() variables contained by the supplied
218       * <code>Matcher</code>}, as well as {@linkplain Matcher#group(int) its
219       * capture groups}.  Capture group variable references begin with
220       * the {@code $} character, so the following orb tag would return a
221       * {@link List} representing the full match:</p>
222       *
223       * <blockquote><code>@{$0}</code></blockquote>
224       *
225       * @param rawMessage the unformatted message as returned by the
226       * {@link #getMessage(List)} method; may be {@code null}
227       *
228       * @param matcher the {@link Matcher} that was used by the {@link
229       * #getMessage(List)} method in producing the {@code rawMessage}
230       * parameter value; may be {@code null}
231       *
232       * @return a formatted version of the supplied {@code rawMessage},
233       * which may simply be the supplied {@code rawMessage} if no formatting
234       * could be performed
235       *
236       * @see TemplateRuntime
237       *
238       * @see #getMessage(List)
239       */
240      protected Object format(final Object rawMessage, final Matcher<T> matcher) {
241        final Object returnValue;
242        if (rawMessage instanceof CharSequence) {
243          final String template = rawMessage.toString();
244          if (template != null && matcher != null) {
245            final Map<Object, Object> variables = new HashMap<Object, Object>();
246            final int groupCount = matcher.groupCount();
247            for (int i = 0; i < groupCount; i++) {
248              variables.put(String.format("$%d", Integer.valueOf(i)), matcher.group(i));
249            }
250            variables.putAll(matcher.getVariables());
251            returnValue = TemplateRuntime.eval(template, variables);
252          } else {
253            returnValue = rawMessage;
254          }
255        } else {
256          returnValue = null;
257        }
258        return returnValue;
259      }

261      /**
262       * Given a {@link List} of {@link Object}s of type {@link
263       * MessageFactory T}, matches that {@link List} against all the
264       * {@link Pattern}s that have been {@linkplain
265       * #addPattern(ResourceBundleKey, Pattern) added} to this {@link
266       * MessageFactory} in insertion order, and, when one of them
267       * {@linkplain Matcher#lookingAt() has a match}, selects and
268       * {@linkplain #format(Object, Matcher) formats} the associated key,
269       * {@linkplain #convert(Object) converts it to a
270       * <code>String</code>} and returns the result.
271       *
272       * <p>This method may return {@code null}.</p>
273       *
274       * @param input the {@link List} of {@link Object}s of type {@link
275       * MessageFactory T} to match; may be {@code null}
276       *
277       * @return a {@linkplain #convert(Object) converted} and {@linkplain
278       * #format(Object, Matcher) formatted} message, or {@code null}
279       *
280       * @see #convert(Object)
281       *
282       * @see #format(Object, Matcher)
283       */
284      public String getMessage(final List<T> input) {
285        final String returnValue;
286        final Selector<T> selector = this.getSelector(input);
287        if (selector == null) {
288          returnValue = null;
289        } else {
290          final ResourceBundleKey key = selector.getKey();
291          if (key == null) {
292            returnValue = this.convert(this.format(null, selector.getMatcher()));
293          } else {
294            returnValue = this.convert(this.format(key.getObject(), selector.getMatcher()));
295          }
296        }    
297        return returnValue;
298      }

300      /**
301       * Converts the supplied {@link Object} into a {@link String}.  The
302       * default implementation of this method returns {@code null} if the
303       * supplied {@link Object} is {@code null}, or the result of
304       * invoking {@link Object#toString()} on it if it is not.
305       *
306       * <p>This method may return {@code null}.</p>
307       *
308       * @param object the {@link Object} to convert; may be {@code null}
309       *
310       * @return a {@link String} that is the result of converting the
311       * supplied {@link Object}, or {@code null}
312       */
313      protected String convert(final Object object) {
314        final String returnValue;
315        if (object == null) {
316          returnValue = null;
317        } else {
318          returnValue = object.toString();
319        }
320        return returnValue;
321      }

323      /**
324       * Calls {@link #getMessage(List)}, and if the return value is {@code
325       * null}, returns the supplied {@code defaultValue} parameter as the
326       * return value instead.
327       *
328       * <p>This method may return {@code null}.</p>
329       *
330       * <p>This method suppresses any {@link MissingResourceException}s
331       * thrown by {@link #getMessage(List)} and returns the supplied
332       * {@code defaultValue} parameter in such cases.</p>
333       *
334       * @return a formatted message, or the {@code defaultValue}
335       * parameter, or {@code null} if the {@code defaultValue} parameter
336       * is {@code null} itself
337       */
338      public String getMessage(final List<T> input, final String defaultValue) {
339        String returnValue = null;
340        try {
341          returnValue = this.getMessage(input);
342        } catch (final MissingResourceException oops) {
343          // TODO: log
344          returnValue = null;
345        }
346        if (returnValue == null) {
347          returnValue = defaultValue;
348        }
349        return returnValue;
350      }

352      /**
353       * Returns a {@link Selector} for the supplied {@link List} of
354       * {@link Object}s of type {@link MessageFactory T}.
355       *
356       * <p>This method may return {@code null}.</p>
357       *
358       * @param input the {@link List} of {@link Object}s of type {@link
359       * MessageFactory T} for which a {@link Selector} will be returned;
360       * may be {@code null}; will be passed to the {@link
361       * Pattern#matcher(List)} method
362       *
363       * @return a {@link Selector}, or {@code null}
364       */
365      final Selector<T> getSelector(final List<T> input) {
366        Selector<T> returnValue = null;
367        if (this.patterns != null && !this.patterns.isEmpty()) {
368          final Set<Entry<ResourceBundleKey, Set<Pattern<T>>>> entrySet = this.patterns.entrySet();
369          if (entrySet != null && !entrySet.isEmpty()) {
370            ENTRY_SET_LOOP:
371            for (final Entry<ResourceBundleKey, Set<Pattern<T>>> entry : entrySet) {
372              if (entry != null) {
373                final ResourceBundleKey key = entry.getKey();
374                if (key != null) {
375                  final Set<Pattern<T>> patterns = entry.getValue();
376                  if (patterns != null && !patterns.isEmpty()) {
377                    for (final Pattern<T> pattern : patterns) {
378                      if (pattern != null) {
379                        final Matcher<T> matcher = pattern.matcher(input);
380                        assert matcher != null;
381                        if (matcher.lookingAt()) {
382                          returnValue = new Selector<T>(key, matcher);
383                          break ENTRY_SET_LOOP;
384                        }
385                      }
386                    }
387                  }
388                }
389              }
390            }
391          }
392        }
393        return returnValue;
394      }

396      /**
397       * A simple tuple that combines a {@link ResourceBundle}, a {@link
398       * String} key and a {@link Matcher}.
399       *
400       * @author <a href="http://about.me/lairdnelson"
401       * target="_parent">Laird Nelson</a>
402       *
403       * @see MessageFactory#getSelector(List)
404       */
405      private static final class Selector<T> {
406    
407        /**
408         * The {@link Selector}'s key.  This field is never {@code null}.
409         */
410        private final ResourceBundleKey key;
411    
412        /**
413         * This {@link Selector}'s {@link Matcher}.  This field is never
414         * {@code null}.
415         */
416        private final Matcher<T> matcher;
417    
418        /**
419         * Creates a new {@link Selector}.
420         *
421         * @param key the key portion of this {@link Selector}; must not
422         * be {@code null}
423         *
424         * @param matcher the {@link Matcher} portion of this {@link
425         * Selector}; must not be {@code null}
426         *
427         * @exception IllegalArgumentException if either parameter is
428         * {@code null}
429         */
430        private Selector(final ResourceBundleKey key, final Matcher<T> matcher) {
431          super();
432          if (key == null) {
433            throw new IllegalArgumentException("key", new NullPointerException("key"));
434          }
435          if (matcher == null) {
436            throw new IllegalArgumentException("matcher", new NullPointerException("matcher"));
437          }
438          this.key = key;
439          this.matcher = matcher;
440        }
441    
442        /**
443         * Returns the key of this {@link Selector}.
444         *
445         * <p>This method never returns {@code null}.</p>
446         *
447         * @return the non-{@code null} key of this {@link Selector}
448         */
449        public final ResourceBundleKey getKey() {
450          return this.key;
451        }
452    
453        /**
454         * Returns the {@link Matcher} of this {@link Selector}.
455         *
456         * <p>This method never returns {@code null}.</p>
457         *
458         * @return the non-{@code null} {@link Matcher} portion of this
459         * {@link Selector}
460         */ 
461        public final Matcher<T> getMatcher() {
462          return this.matcher;
463        }
464    
465        /**
466         * Returns a hashcode for this {@link Selector}.
467         *
468         * @return a hashcode for this {@link Selector}
469         */
470        @Override
471        public int hashCode() {
472          int result = 17;
473          
474          final Object key = this.getKey();
475          int c;
476          if (key == null) {
477            c = 0;
478          } else {
479            c = key.hashCode();
480          }
481          result = 37 * result + c;
482    
483          final Object matcher = this.getMatcher();
484          if (matcher == null) {
485            c = 0;
486          } else {
487            c = matcher.hashCode();
488          }
489          result = 37 * result + c;
490    
491          return result;
492        }
493    
494        /**
495         * Returns {@code true} if the supplied {@link Object} is equal to
496         * this {@link Selector}; {@code false} otherwise.
497         *
498         * @param other the {@link Object} to compare; may be {@code null}
499         *
500         * @return {@code true} if the supplied {@link Object} is equal to
501         * this {@link Selector}; {@code false} otherwise
502         */
503        @Override
504        public boolean equals(final Object other) {
505          if (other == this) {
506            return true;
507          } else if (other != null && this.getClass().equals(other.getClass())) {
508            final Selector<?> him = (Selector<?>)other;
509            final Object key = this.getKey();
510            if (key == null) {
511              if (him.getKey() != null) {
512                return false;
513              }
514            } else if (!key.equals(him.getKey())) {
515              return false;
516            }
517    
518            final Object matcher = this.getMatcher();
519            if (matcher == null) {
520              if (him.getMatcher() != null) {
521                return false;
522              }
523            } else if (!matcher.equals(him.getMatcher())) {
524              return false;
525            }
526    
527            return true;
528          } else {
529            return false;
530          }
531        }
532    
533      }
