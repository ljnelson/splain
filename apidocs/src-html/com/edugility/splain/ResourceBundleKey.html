193      /**
194       * Creates a new {@link ResourceBundleKey}.
195       *
196       * @param resourceBundleName a fully-qualified class name
197       * identifying a {@link ResourceBundle}; may be {@code null}
198       *
199       * @param bundleLoader a {@link ClassLoader} for passing to the
200       * {@link ResourceBundle#getBundle(String, Locale, ClassLoader,
201       * Control)} method; may be {@code null} in which case the return
202       * value of {@link Thread#getContextClassLoader()
203       * Thread.currentThread().getContextClassLoader()} will be used
204       * instead
205       *
206       * @param control a {@link Control} to use for instantiating {@link
207       * ResourceBundle}s; may be {@code null} in which case the return
208       * value of the {@link Control#getControl(List)}
209       * method&mdash;supplied with the {@link Control#FORMAT_DEFAULT}
210       * constant&mdash;will be used instead
211       *
212       * @param key a {@link String} identifying a resource within a
213       * {@link ResourceBundle} identified by the supplied {@code
214       * resourceBundleName} parameter, or, if that parameter is {@code
215       * null} or {@linkplain String#isEmpty() empty}, the value to be
216       * returned from the {@link #getObject(Locale)} method; must not be
217       * {@code null}
218       *
219       * @exception IllegalArgumentException if {@code key} is {@code
220       * null}
221       */
222      public ResourceBundleKey(final String resourceBundleName, ClassLoader bundleLoader, final Control control, final String key) {
223        super();
224        if (key == null) {
225          throw new IllegalArgumentException("key", new NullPointerException("key"));
226        }
227        if (control == null) {
228          this.control = Control.getControl(Control.FORMAT_DEFAULT);
229        } else {
230          this.control = control;
231        }
232        if (bundleLoader == null) {
233          bundleLoader = Thread.currentThread().getContextClassLoader();
234          if (bundleLoader == null) {
235            bundleLoader = this.getClass().getClassLoader();
236          }
237        }
238        assert bundleLoader != null;
239        this.bundleLoader = bundleLoader;
240        this.resourceBundleName = resourceBundleName;
241        this.key = key;
242      }

264        } else {
265          returnValue = this.control;
266        }
267        return returnValue;

270      /**
271       * @exception MissingResourceException
272       */
273      private final ResourceBundle getResourceBundle(Locale locale) {
274        final ResourceBundle returnValue;
275        final String resourceBundleName = this.getResourceBundleName();
276        if (resourceBundleName == null) {
277          returnValue = null;
278        } else {
279          if (locale == null) {
280            locale = Locale.getDefault();
281          }
282          assert locale != null;
283          Control control = this.getControl();
284          if (control == null) {
285            control = Control.getControl(Control.FORMAT_DEFAULT);
286          }
287          assert control != null;
288          assert this.bundleLoader != null;
289          returnValue = ResourceBundle.getBundle(resourceBundleName, locale, this.bundleLoader, control);
290        }
291        return returnValue;
292      }

326      /**
327       * Attempts to {@linkplain ResourceBundle#getBundle(String, Locale,
328       * ClassLoader, Control) load} the {@link ResourceBundle} identified
329       * by this {@link ResourceBundleKey} and to return the resource in
330       * that bundle identified by this {@link ResourceBundleKey}.
331       *
332       * <p>If this {@link ResourceBundleKey} was constructed with a
333       * {@code null} resource bundle name, then the key that was supplied
334       * at construction time is simply returned as is.</p>
335       *
336       * <p>This method may return {@code null} in rare edge cases
337       * only.</p>
338       *
339       * @param locale the {@link Locale} for which an {@link Object}
340       * should be retrieved; may be {@code null} in which case the return
341       * value of the {@link Locale#getDefault()} method will be used
342       * instead
343       *
344       * @return a resource, or {@code null}
345       *
346       * @exception MissingResourceException if either an appropriate
347       * {@link ResourceBundle} could not be {@linkplain
348       * ResourceBundle#getBundle(String, Locale, ClassLoader, Control)
349       * loaded} or if the relevant resource within that {@link
350       * ResourceBundle} does not {@linkplain
351       * ResourceBundle#containsKey(String) exist}
352       */
353      public final Object getObject(final Locale locale) {
354        Object returnValue = null;
355        final ResourceBundle rb = this.getResourceBundle(locale);
356        final String key = this.getKey();
357        if (rb != null && key != null) {
358          returnValue = rb.getObject(key);
359        } else {
360          returnValue = key;
361        }
362        return returnValue;
363      }

366       * Returns a hash code for this {@link ResourceBundleKey}.
367       *
368       * @return a hash code for this {@link ResourceBundleKey}
369       */
370      @Override
371      public final int hashCode() {
372        int result = 17;
373    
374        final Object control = this.getControl();
375        if (control != null) {
376          result = result * 37 + control.hashCode();
377        }
378    
379        final Object resourceBundleName = this.getResourceBundleName();
380        if (resourceBundleName != null) {
381          result = result * 37 + resourceBundleName.hashCode();
382        }
383    
384        final Object key = this.getKey();
385        if (key != null) {
386          result = result * 37 + key.hashCode();
387        }
388    
389        final Object bundleLoader = this.getBundleLoader();
390        if (bundleLoader != null) {
391          result = result * 37 + bundleLoader.hashCode();
392        }
393    
394        return result;
395      }

398       * Returns {@code true} if the supplied {@link Object} is equal to
399       * this {@link ResourceBundleKey}; {@code false} otherwise.
400       *
401       * @param other the {@link Object} to test; may be {@code null}
402       *
403       * @return {@code true} if and only if the supplied {@link Object}

406       */
407      @Override
408      public final boolean equals(final Object other) {
409        if (other == this) {
410          return true;
411        } else if (other != null && this.getClass().equals(other.getClass())) {
412          final ResourceBundleKey him = (ResourceBundleKey)other;
413    
414          final Object key = this.getKey();
415          if (key == null) {
416            if (him.getKey() != null) {
417              return false;
418            }
419          } else if (!key.equals(him.getKey())) {
420            return false;
421          }
422    
423          final Object resourceBundleName = this.getResourceBundleName();
424          if (resourceBundleName == null) {
425            if (him.getResourceBundleName() != null) {
426              return false;
427            }
428          } else if (!resourceBundleName.equals(him.getResourceBundleName())) {
429            return false;
430          }
431    
432          final Object control = this.getControl();
433          if (control == null) {
434            if (him.getControl() != null) {
435              return false;
436            }
437          } else if (!control.equals(him.getControl())) {
438            return false;
439          }
440    
441          final Object bundleLoader = this.getBundleLoader();
442          if (bundleLoader == null) {
443            if (him.getBundleLoader() != null) {
444              return false;
445            }
446          } else if (!bundleLoader.equals(him.getBundleLoader())) {
447            return false;
448          }
449    
450          return true;
451        } else {
452          return false;
453        }
454      }

456      /**
457       * Returns a non-{@code null} {@link String} representation of this
458       * {@link ResourceBundleKey}.  The representation is of the key
459       * itself, <em>not</em> of the return value of the {@link
460       * #getObject(Locale)} method.
461       *
462       * <p>This method never returns {@code null}.</p>
463       *
464       * <p>The format of the {@link String} representation returned by
465       * this method may change at any time.</p>
466       *
467       * @return a non-{@code null} {@link String}
468       */
469      @Override
470      public final String toString() {
471        final StringBuilder sb = new StringBuilder();
472        final Object resourceBundleName = this.getResourceBundleName();
473        if (resourceBundleName != null) {
474          sb.append(resourceBundleName);
475          sb.append("/");
476        }
477        final Object key = this.getKey();
478        if (key != null) {
479          sb.append(key);
480        }
481        return sb.toString();
482      }

501         * The {@link State} the parser is in to start.
502         */

511    
512        /**
513         * The {@link State} the parser is in when it is accumulating
514         * {@link ResourceBundle} name characters.

516        BUNDLE_NAME,

518        /**

520         * accumulating key characters.
521         */
522        BUNDLE_KEY_START,

524        /**
525         * The {@link State} the parser is in when it is accumulating key
526         * characters.
527         */
528        BUNDLE_KEY

556      /**
557       * Returns a new {@link ResourceBundleKey} that is appropriate for
558       * the supplied {@code key}.
559       *
560       * <p>This method never returns {@code null}.</p>
561       *
562       * <p>The supplied {@code key} is checked to see if it is {@code
563       * null} or {@linkplain String#isEmpty() empty}.  If so, an {@link
564       * IllegalArgumentException} is thrown.</p>
565       *
566       * <p>The non-{@linkplain String#isEmpty() empty} key is then parsed
567       * into a maximum of two components:</p>
568       *
569       * <ol>
570       *
571       * <li>A <em>bundle name</em>.  A bundle name is a {@link String} of
572       * the form normally supplied to {@link
573       * ResourceBundle#getBundle(String, Locale, ClassLoader,
574       * Control)}&mdash;that is, a fully-qualified class name whose
575       * segments are separated with periods ("{@code .}").  This method
576       * uses {@link Character#isJavaIdentifierStart(char)} and its ilk to
577       * ensure that the bundle name conforms to the Java Language
578       * Specification requirements for class naming.</li>
579       *
580       * <li>A <em>bundle key</em> component, separated from the bundle
581       * name with a solidus ("{@code /}"; the solidus is part of neither
582       * the bundle name nor the bundle key).  A bundle key is an
583       * arbitrary {@link String} that is legal for supplying to the
584       * {@link ResourceBundle#getObject(String)} method.  Leading and
585       * trailing whitespace is removed.</li>
586       *
587       * </ol>
588       *
589       * <p>If the bundle name after this parsing and normalization does
590       * not conform lexically to the requirements of a Java class name,
591       * then the entire supplied {@code key} is treated as a simple
592       * {@link String}, and no further attempts will be made to resolve
593       * it against a {@link ResourceBundle}.</p>
594       *
595       * <p>If the bundle key after this parsing and normalization is
596       * {@linkplain String#isEmpty() empty}, then an {@code
597       * IllegalArgumentException} is thrown.</p>
598       *
599       * <p>Examples:</p>
600       *
601       * <blockquote><dl>
602       *
603       * <dt>{@code com.foobar.ErrorMessages/noSuchElement}</dt>
604       *
605       * <dd>A {@link ResourceBundleKey} whose bundle name is {@code
606       * com.foobar.ErrorMessages} and whose bundle key is {@code
607       * noSuchElement}.</dd>
608       *
609       * <dt>{@code com.foobar.ErrorMessages /noSuchElement}</dt>
610       *
611       * <dd>A {@link ResourceBundleKey} whose bundle name is {@code null}
612       * and whose bundle key is {@code com.foobar.ErrorMessages
613       * /noSuchElement}.  This is because there is whitespace in the
614       * bundle name, so the entire key is treated as a simple {@link
615       * String}.</dd>
616       *
617       * <dt>{@code There was no file found with that name.}</dt>
618       *
619       * <dd>A {@link ResourceBundleKey} that wraps a simple {@link
620       * String} consisting of the text given.</dd>
621       *
622       * <dt>{@code com.foobar.ErrorMessages/  noSuchElement}</dt>
623       *
624       * <dd>A {@link ResourceBundleKey} whose bundle name is {@code
625       * com.foobar.ErrorMessages} and whose bundle key is {@code
626       * noSuchElement}.</dd>
627       *
628       * </dl></blockquote>
629       *
630       * @param bundleLoader the {@link ClassLoader} to pass (ultimately)
631       * to the {@link ResourceBundle#getBundle(String, Locale,
632       * ClassLoader, Control)} method; if {@code null} then the return
633       * value of {@link Thread#getContextClassLoader()
634       * Thread.currentThread().getContextClassLoader()} will be used
635       * instead
636       *
637       * @param control the {@link Control} to use when {@linkplain
638       * ResourceBundle#getBundle(String, Locale, ClassLoader, Control)
639       * loading <code>ResourceBundle</code>s}; if {@code null} then the
640       * return value of {@link Control#getControl(List)
641       * Control.getControl(Control.FORMAT_DEFAULT)} is used instead
642       *
643       * @param key the key to parse; must not be {@code null}
644       *
645       * @return a new {@link ResourceBundleKey}; never {@code null}
646       *
647       * @exception IllegalArgumentException if {@code key} is {@code null}
648       *
649       * @exception MissingResourceException if a {@link ResourceBundle}
650       * could not be loaded or a resource within a {@link ResourceBundle}
651       * could not be found
652       */
653      public static final ResourceBundleKey valueOf(final ClassLoader bundleLoader, final Control control, String key) {
654        if (key == null) {
655          throw new IllegalArgumentException("key", new NullPointerException("key"));
656        } else if (key.isEmpty()) {
657          throw new IllegalArgumentException("key.isEmpty()");
658        }
659    
660        State state = State.START;
661    
662        String bundleName = null;
663        String bundleKey = null;
664    
665        final char[] keyChars = key.toCharArray();
666        assert keyChars != null;
667        final int keyCharsLength = keyChars.length;
668        assert keyCharsLength > 0;
669    
670        final StringBuilder sb = new StringBuilder();
671    
672        for (int i = 0; i < keyCharsLength; i++) {
673          final char c = keyChars[i];
674          switch (state) {
675    
676            // START
677          case START:
678            switch (c) {
679            case '/':
680              throw new IllegalArgumentException(String.format("Malformed key: %s", key));
681            default:
682              if (Character.isJavaIdentifierStart(c)) {
683                state = State.BUNDLE_NAME;
684                sb.append(c);
685              } else if (!Character.isWhitespace(c)) {
686                sb.append(c);
687                bundleName = null;
688                state = State.BUNDLE_KEY;
689              }
690              break;
691            }
692            break;
693    
694            // BUNDLE_NAME_SEGMENT_START
695          case BUNDLE_NAME_SEGMENT_START:
696            sb.append(c);
697            if (Character.isJavaIdentifierStart(c)) {
698              state = State.BUNDLE_NAME;
699            } else {
700              bundleName = null;
701              state = State.BUNDLE_KEY;
702            }
703            break;
704    
705            // BUNDLE_NAME
706          case BUNDLE_NAME:
707            switch (c) {
708            case '.':
709              sb.append('.');
710              state = State.BUNDLE_NAME_SEGMENT_START;
711              break;
712            case '/':
713              if (sb.length() <= 0) {
714                bundleName = null;
715              } else {
716                bundleName = sb.toString();
717              }
718              sb.setLength(0);
719              state = State.BUNDLE_KEY_START;
720              break;
721            default:
722              sb.append(c);
723              if (!Character.isJavaIdentifierPart(c)) {
724                bundleName = null;
725                state = State.BUNDLE_KEY;
726              }
727              break;
728            }
729            break;
730    
731            // BUNDLE_KEY_START
732          case BUNDLE_KEY_START:
733            if (!Character.isWhitespace(c)) {
734              sb.append(c);
735              state = State.BUNDLE_KEY;
736            }
737            break;
738    
739            // BUNDLE_KEY
740          case BUNDLE_KEY:
741            sb.append(c);
742            break;
743    
744          default:
745            throw new IllegalStateException("Unexpected state: " + state);
746          }
747        }
748    
749        switch (state) {
750        case BUNDLE_KEY:
751          bundleKey = sb.toString();
752          sb.setLength(0);
753          break;
754        case BUNDLE_NAME:
755          bundleName = null;
756          bundleKey = sb.toString();
757          sb.setLength(0);
758          break;
759        default:
760          throw new IllegalArgumentException(String.format("Malformed key: %s", key));
761        }
762    
763        assert bundleKey != null;
764        bundleKey = bundleKey.trim();
765        if (bundleKey.isEmpty()) {
766          throw new IllegalArgumentException(String.format("Malformed key: %s", key));
767        }
768    
769        final ResourceBundleKey returnValue = new ResourceBundleKey(bundleName, bundleLoader, control, bundleKey);
770        return returnValue;
771      }
