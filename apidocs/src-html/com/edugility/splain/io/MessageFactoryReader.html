079       * A state that the internal parser can be in.
080       *
081       * @author <a href="http://about.me/lairdnelson"
082       * target="_parent">Laird Nelson</a>
083       */
084      private enum State {
085    
086        /**
087         * A normal state.  The parser will be expecting comment lines or
088         * lines describing patterns.
089         */
090        NORMAL,

092        /**
093         * A state in which the parser will be looking for patterns or
094         * comments.
095         */
096        MATCHERS,
097    
098        /**
099         * A state in which the parser will attempt to read the message
100         * key that is affiliated with a set of patterns.
101         */
102        MESSAGE
103      }

161      /**
162       * Creates a new {@link MessageFactoryReader}.
163       *
164       * @param resourceName the name of a resource to be loaded via the
165       * {@link ClassLoader#getResource(String)} method; must not be
166       * {@code null}
167       *
168       * @param defaultResourceBundle a {@link ResourceBundle} to use to
169       * resolve relative keys; may be {@code null}
170       *
171       * @param locale a {@link Locale} to use when {@linkplain
172       * ResourceBundle#getBundle(String, Locale, ClassLoader,
173       * ResourceBundle.Control) loading <code>ResourceBundle</code>s}; if
174       * {@code null} then {@link Locale#getDefault() Locale.getDefault()}
175       * will be used instead
176       *
177       * @param rbControl a {@link Control} to use when {@linkplain
178       * ResourceBundle#getBundle(String, Locale, ClassLoader,
179       * ResourceBundle.Control) loading <code>ResourceBundle</code>s}; if
180       * {@code null} then {@link Control#getControl(List)
181       * Control.getControl(Control.FORMAT_DEFAULT)} will be used instead
182       *
183       * @exception IllegalArgumentException if {@code resourceName} is
184       * {@code null} or identifies a resource that could not be opened
185       *
186       * @exception IOException if an input or output error occurs
187       */
188      public MessageFactoryReader(final String resourceName, final ResourceBundle defaultResourceBundle, final Locale locale, final Control rbControl) throws IOException {
189        super();
190        if (resourceName == null) {
191          throw new IllegalArgumentException("resourceName", new NullPointerException("resourceName"));
192        }
193        final InputStream resource;
194        ClassLoader cl = Thread.currentThread().getContextClassLoader();
195        if (cl == null) {
196          if (resourceName.startsWith("/")) {
197            resource = this.getClass().getResourceAsStream(resourceName);
198          } else {
199            resource = this.getClass().getResourceAsStream(String.format("/%s", resourceName));
200          }
201        } else {
202          resource = cl.getResourceAsStream(resourceName);
203        }
204        if (resource == null) {
205          throw new IllegalArgumentException("resourceName", new IllegalStateException("resource not found"));
206        }
207        this.reader = new LineNumberReader(new BufferedReader(new InputStreamReader(resource)));
208        if (locale == null) {
209          this.locale = Locale.getDefault();
210        } else {
211          this.locale = locale;
212        }
213        assert this.locale != null;
214        if (rbControl == null) {
215          this.control = Control.getControl(Control.FORMAT_DEFAULT);
216        } else {
217          this.control = rbControl;
218        }
219        this.defaultResourceBundle = defaultResourceBundle;
220      }

250      /**
251       * Creates a new {@link MessageFactoryReader}.
252       *
253       * @param reader the {@link Reader} to read from; must not be {@code
254       * null}
255       *
256       * @param defaultResourceBundle the {@link ResourceBundle} to use to
257       * resolve unqualified keys against; may be {@code null}
258       *
259       * @param locale the {@link Locale} to use for localization when
260       * loading new {@link ResourceBundle}s; if {@code null} then the
261       * {@linkplain Locale#getDefault() default <code>Locale</code>} will
262       * be used instead
263       *
264       * @param rbControl the {@link Control} to use when loading new
265       * {@link ResourceBundle}s; if {@code null} then {@link
266       * Control#getControl(List)
267       * Control.getControl(Control.FORMAT_DEFAULT)} will be used instead
268       *
269       * @exception IllegalArgumentException if {@code reader} is {@code
270       * null}
271       */
272      public MessageFactoryReader(final Reader reader, final ResourceBundle defaultResourceBundle, final Locale locale, final Control rbControl) {
273        super();
274        if (reader == null) {
275          throw new IllegalArgumentException("reader", new NullPointerException("reader"));
276        }
277        if (reader instanceof LineNumberReader) {
278          this.reader = (LineNumberReader)reader;
279        } else if (reader instanceof BufferedReader) {
280          this.reader = new LineNumberReader(reader);
281        } else {
282          this.reader = new LineNumberReader(new BufferedReader(reader));
283        }
284        if (locale == null) {
285          this.locale = Locale.getDefault();
286        } else {
287          this.locale = locale;
288        }
289        if (rbControl == null) {
290          this.control = Control.getControl(Control.FORMAT_DEFAULT);
291        } else {
292          this.control = rbControl;
293        }
294        this.defaultResourceBundle = defaultResourceBundle;
295      }

331      /**
332       * Reads from this {@link MessageFactoryReader}'s {@linkplain
333       * #MessageFactoryReader(Reader) affiliated <tt>Reader</tt>} and
334       * builds a {@link MessageFactoryReader} from the results.
335       *
336       * <p>To avoid resource leaks, this {@link MessageFactoryReader}
337       * should be {@linkplain #close() closed} after this method
338       * completes normally.  This method does not call the {@link
339       * #close()} method itself.</p>
340       *
341       * @param <T> the type of objects new {@link MessageFactory}
342       * instances produced from this method will work with
343       *
344       * @return a new {@link MessageFactory}; never {@code null}
345       *
346       * @exception IOException if an input or output error occurs
347       *
348       * @exception IllegalStateException if parsing could not be
349       * completed
350       *
351       * @exception ParseException if the source code could not be parsed
352       */
353      public <T> MessageFactory<T> read() throws IOException, ParseException {
354        assert this.reader != null;
355        final MessageFactory<T> mf = new MessageFactory<T>();
356        State state = State.NORMAL;
357        final Set<Pattern<T>> patterns = new LinkedHashSet<Pattern<T>>();
358        StringBuilder message = null;
359        final ResourceBundle defaultResourceBundle = this.getDefaultResourceBundle();
360        String line;
361        while ((line = this.reader.readLine()) != null) {
362          line = line.trim();
363    
364          switch (state) {
365    
366            // NORMAL
367          case NORMAL:
368            if (line.isEmpty() || line.startsWith("#")) {
369              break;
370            } else if (line.startsWith("--")) {
371              throw new IllegalStateException("\"--\" is not permitted here at line " + reader.getLineNumber());
372            } else {
373              state = State.MATCHERS;
374              patterns.add(Pattern.<T>compile(line));
375              break;
376            }
377            // end NORMAL
378    
379    
380            // MATCHERS
381          case MATCHERS:
382            if (line.isEmpty()) {
383              throw new IllegalStateException("An empty line is not permitted here at line " + reader.getLineNumber());
384            } else if (line.startsWith("--")) {
385              state = State.MESSAGE;
386            } else if (!line.startsWith("#")) {
387              patterns.add(Pattern.<T>compile(line));
388            }
389            break;
390            // end MATCHERS
391    
392    
393            // MESSAGE
394          case MESSAGE:
395            if (line.isEmpty()) {
396              if (message != null) {
397                assert patterns != null;
398                assert !patterns.isEmpty();
399                addPatterns(mf, defaultResourceBundle, message.toString(), patterns);
400                patterns.clear();
401                message = null;
402                state = State.NORMAL;
403              }
404            } else {
405              if (message == null) {
406                message = new StringBuilder();
407              } else {
408                message.append(LS);
409              }
410              message.append(line);
411            }
412            break;
413            // end MESSAGE
414    
415    
416          default:
417            throw new IllegalStateException("Unexpected state: " + state);
418          }
419    
420          if (message != null && !patterns.isEmpty()) {
421            addPatterns(mf, defaultResourceBundle, message.toString(), patterns);
422            patterns.clear();
423            message = null;
424          }
425    
426        }
427        return mf;
428      }

430      /**
431       * {@linkplain MessageFactory#addPatterns(ResourceBundleKey, Set)
432       * Adds} the supplied {@link Set} of {@link Pattern}s to the
433       * supplied {@link MessageFactory} under a new {@link
434       * ResourceBundleKey} {@linkplain
435       * ResourceBundleKey#valueOf(ResourceBundle, Locale,
436       * ResourceBundle.Control, String) constructed} from the supplied
437       * parameters.
438       *
439       * @param mf the {@link MessageFactory} to which to add {@link
440       * Pattern}s; must not be {@code null}
441       *
442       * @param defaultResourceBundle the {@linkplain
443       * ResourceBundleKey#getResourceBundle() <code>ResourceBundle</code>
444       * component} of a {@link ResourceBundleKey}; may be {@code null}
445       *
446       * @param message the "key" component of a {@link
447       * ResourceBundleKey}; must not be {@code null}
448       *
449       * @param patterns a non-{@code null} {@link Set} of {@link
450       * Pattern}s to add; must not be {@linkplain Collection#isEmpty()
451       * empty}
452       *
453       * @exception IllegalArgumentException if {@code mf}, {@code
454       * message} or {@code patterns} is {@code null}, or if {@code
455       * patterns} is {@linkplain Collections#isEmpty() emtpy}
456       *
457       * @exception MissingResourceException if the {@link
458       * ResourceBundleKey#valueOf(ResourceBundle, Locale,
459       * ResourceBundle.Control, String)} method throws a {@link
460       * MissingResourceException}
461       *
462       * @see ResourceBundleKey#valueOf(ResourceBundle, Locale,
463       * ResourceBundle.Control, String)
464       *
465       * @see MessageFactory#addPatterns(ResourceBundleKey, Set)
466       */
467      private final <T> void addPatterns(final MessageFactory<T> mf, final ResourceBundle defaultResourceBundle, final String message, final Set<Pattern<T>> patterns) {
468        if (mf == null) {
469          throw new IllegalArgumentException("mf", new NullPointerException("mf"));
470        }
471        if (message == null) {
472          throw new IllegalArgumentException("message", new NullPointerException("message"));
473        }
474        if (patterns == null) {
475          throw new IllegalArgumentException("patterns", new NullPointerException("patterns"));
476        }
477        if (patterns.isEmpty()) {
478          throw new IllegalArgumentException("patterns", new IllegalStateException("patterns.isEmpty()"));
479        }
480        final ResourceBundleKey rbKey = ResourceBundleKey.valueOf(defaultResourceBundle, this.getLocale(), this.getControl(), message);
481        assert rbKey != null;
482        mf.addPatterns(rbKey, patterns);
483      }

485      /**
486       * Closes this {@link MessageFactoryReader} so that it cannot be
487       * used again.  Once this method has been called, then the {@link
488       * #read()} method will throw an {@link IOException} whenever it is
489       * invoked.
490       *
491       * @exception IOException if an error was encountered during closing
492       */
493      @Override
494      public void close() throws IOException {
495        if (this.reader != null) {
496          this.reader.close();
497        }
498      }
