365      public MessageFactoryReader(final String resourceName, final ResourceBundle defaultResourceBundle, final Locale locale, final Control rbControl) throws IOException {
366        super();
367        if (resourceName == null) {
368          throw new IllegalArgumentException("resourceName", new NullPointerException("resourceName"));
369        }
370        final InputStream resource;
371        ClassLoader cl = Thread.currentThread().getContextClassLoader();
372        if (cl == null) {
373          if (resourceName.startsWith("/")) {
374            resource = this.getClass().getResourceAsStream(resourceName);
375          } else {
376            resource = this.getClass().getResourceAsStream(String.format("/%s", resourceName));
377          }
378        } else {
379          resource = cl.getResourceAsStream(resourceName);
380        }
381        if (resource == null) {
382          throw new IllegalArgumentException("resourceName", new IllegalStateException("resource not found"));
383        }
384        this.reader = new LineNumberReader(new BufferedReader(new InputStreamReader(resource)));
385        if (locale == null) {
386          this.locale = Locale.getDefault();
387        } else {
388          this.locale = locale;
389        }
390        assert this.locale != null;
391        if (rbControl == null) {
392          this.control = Control.getControl(Control.FORMAT_DEFAULT);
393        } else {
394          this.control = rbControl;
395        }
396        this.defaultResourceBundle = defaultResourceBundle;
397      }

508      /**
509       * Reads from this {@link MessageFactoryReader}'s {@linkplain
510       * #MessageFactoryReader(Reader) affiliated <tt>Reader</tt>} and
511       * builds a {@link MessageFactoryReader} from the results.
512       *
513       * <p>To avoid resource leaks, this {@link MessageFactoryReader}
514       * should be {@linkplain #close() closed} after this method
515       * completes normally.  This method does not call the {@link
516       * #close()} method itself.</p>
517       *
518       * @param <T> the type of objects new {@link MessageFactory}
519       * instances produced from this method will work with
520       *
521       * @return a new {@link MessageFactory}; never {@code null}
522       *
523       * @exception IOException if an input or output error occurs
524       *
525       * @exception IllegalStateException if parsing could not be
526       * completed
527       *
528       * @exception ParseException if the source code could not be parsed
529       */
530      public <T> MessageFactory<T> read() throws IOException, ParseException {
531        assert this.reader != null;
532        final MessageFactory<T> mf = new MessageFactory<T>();
533        State state = State.NORMAL;
534        final Set<Pattern<T>> patterns = new LinkedHashSet<Pattern<T>>();
535        StringBuilder message = null;
536        final ResourceBundle defaultResourceBundle = this.getDefaultResourceBundle();
537        String line;
538        while ((line = this.reader.readLine()) != null) {
539          line = line.trim();
540    
541          switch (state) {
542    
543            // NORMAL
544          case NORMAL:
545            if (line.isEmpty() || line.startsWith("#")) {
546              break;
547            } else if (line.startsWith("--")) {
548              throw new IllegalStateException("\"--\" is not permitted here at line " + reader.getLineNumber());
549            } else {
550              state = State.MATCHERS;
551              patterns.add(Pattern.<T>compile(line));
552              break;
553            }
554            // end NORMAL
555    
556    
557            // MATCHERS
558          case MATCHERS:
559            if (line.isEmpty()) {
560              throw new IllegalStateException("An empty line is not permitted here at line " + reader.getLineNumber());
561            } else if (line.startsWith("--")) {
562              state = State.MESSAGE;
563            } else if (!line.startsWith("#")) {
564              patterns.add(Pattern.<T>compile(line));
565            }
566            break;
567            // end MATCHERS
568    
569    
570            // MESSAGE
571          case MESSAGE:
572            if (line.isEmpty()) {
573              if (message != null) {
574                assert patterns != null;
575                assert !patterns.isEmpty();
576                addPatterns(mf, defaultResourceBundle, message.toString(), patterns);
577                patterns.clear();
578                message = null;
579                state = State.NORMAL;
580              }
581            } else {
582              if (message == null) {
583                message = new StringBuilder();
584              } else {
585                message.append(LS);
586              }
587              message.append(line);
588            }
589            break;
590            // end MESSAGE
591    
592    
593          default:
594            throw new IllegalStateException("Unexpected state: " + state);
595          }
596    
597          if (message != null && !patterns.isEmpty()) {
598            addPatterns(mf, defaultResourceBundle, message.toString(), patterns);
599            patterns.clear();
600            message = null;
601          }
602    
603        }
604        return mf;
605      }

607      /**
608       * {@linkplain MessageFactory#addPatterns(ResourceBundleKey, Set)
609       * Adds} the supplied {@link Set} of {@link Pattern}s to the
610       * supplied {@link MessageFactory} under a new {@link
611       * ResourceBundleKey} {@linkplain
612       * ResourceBundleKey#valueOf(ResourceBundle, Locale,
613       * ResourceBundle.Control, String) constructed} from the supplied
614       * parameters.
615       *
616       * @param mf the {@link MessageFactory} to which to add {@link
617       * Pattern}s; must not be {@code null}
618       *
619       * @param defaultResourceBundle the {@linkplain
620       * ResourceBundleKey#getResourceBundle() <code>ResourceBundle</code>
621       * component} of a {@link ResourceBundleKey}; may be {@code null}
622       *
623       * @param message the "key" component of a {@link
624       * ResourceBundleKey}; must not be {@code null}
625       *
626       * @param patterns a non-{@code null} {@link Set} of {@link
627       * Pattern}s to add; must not be {@linkplain Collection#isEmpty()
628       * empty}
629       *
630       * @exception IllegalArgumentException if {@code mf}, {@code
631       * message} or {@code patterns} is {@code null}, or if {@code
632       * patterns} is {@linkplain Collections#isEmpty() emtpy}
633       *
634       * @exception MissingResourceException if the {@link
635       * ResourceBundleKey#valueOf(ResourceBundle, Locale,
636       * ResourceBundle.Control, String)} method throws a {@link
637       * MissingResourceException}
638       *
639       * @see ResourceBundleKey#valueOf(ResourceBundle, Locale,
640       * ResourceBundle.Control, String)
641       *
642       * @see MessageFactory#addPatterns(ResourceBundleKey, Set)
643       */
644      private final <T> void addPatterns(final MessageFactory<T> mf, final ResourceBundle defaultResourceBundle, final String message, final Set<Pattern<T>> patterns) {
645        if (mf == null) {
646          throw new IllegalArgumentException("mf", new NullPointerException("mf"));
647        }
648        if (message == null) {
649          throw new IllegalArgumentException("message", new NullPointerException("message"));
650        }
651        if (patterns == null) {
652          throw new IllegalArgumentException("patterns", new NullPointerException("patterns"));
653        }
654        if (patterns.isEmpty()) {
655          throw new IllegalArgumentException("patterns", new IllegalStateException("patterns.isEmpty()"));
656        }
657        final ResourceBundleKey rbKey = ResourceBundleKey.valueOf(defaultResourceBundle, this.getLocale(), this.getControl(), message);
658        assert rbKey != null;
659        mf.addPatterns(rbKey, patterns);
660      }

665       * #read()} method will throw an {@link IOException} whenever it is
666       * invoked.
667       *
668       * @exception IOException if an error was encountered during closing
669       */
670      @Override
671      public void close() throws IOException {
672        if (this.reader != null) {
673          this.reader.close();
674        }
